// Firebase Storage Security Rules
// Add these rules to your Firebase Storage security rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow authenticated users to upload and manage company logos
    match /company/logo/{fileName} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to upload and manage job attachments and files
    match /jobs/{jobId}/attachments/{fileName} {
      allow read, write: if request.auth != null;
    }
    
    // Statement of Conformity reference PDF (stored on job as serviceInformation.statementOfConformityReferencePdf)
    match /jobs/{jobId}/service-information/statement-of-conformity/{fileName} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to manage all job-related files (for deletion)
    match /jobs/{jobId}/{allPaths=**} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to upload and manage equipment attachments
    // Files are scoped to: jobs/{jobId}/equipment/{equipmentIndex}/attachments/{fileName}
    match /jobs/{jobId}/equipment/{equipmentIndex}/attachments/{fileName} {
      // Only authenticated users with job access can read
      allow read: if request.auth != null;
      // Only authenticated users can upload (permission check done in app)
      allow write: if request.auth != null;
    }
    
    // Allow authenticated users to upload and manage customer files
    match /customers/{customerId}/files/{fileName} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to upload and manage user avatars
    match /users/{userId}/avatar/{fileName} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Default rule: deny all other access
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
